130 ActuatorState evaluate(const SensorData& data) {
131 ActuatorState state;
132 state.waterPump = 0;
133 state.fan = 0;
134 state.leds = 0;
135
136 // Evaluar humedad del suelo
137 if (data.soilMoisture < minSoilMoisture) {
138 state.waterPump = 1;
139 alertLowSoilMoisture = true;
140 } else {
141 alertLowSoilMoisture = false;
142 }
143
144 if (data.soilMoisture > maxSoilMoisture) {
145 alertHighSoilMoisture = true;
146 } else {
147 alertHighSoilMoisture = false;
148 }
149
150 // Evaluar temperatura
151 if (data.temperature < minTemperature) {
152 alertLowTemperature = true;
153 } else {
154 alertLowTemperature = false;
155 }
156
157 if (data.temperature > maxTemperature) {
158 alertHighTemperature = true;
159 state.fan = 1;
160 } else {
161 alertHighTemperature = false;
162 }
163
164 // Evaluar CO2
165 if (data.co2ppm > maxCO2) {
166 alertCO2 = true;
167 state.fan = 1;
168 } else {
169 alertCO2 = false;
170 }
171
172 // Evaluar luz
173 if (data.light < minLight) {
174 alertLight = true;
175 state.leds = 1;
176 } else {
177 alertLight = false;
178 }
179
180 // Evaluar voltaje
181 if (data.voltage < minVoltage) {
182 alertVoltage = true;
183 } else {
184 alertVoltage = false;
185 }
186 return state;
187
188 // Evaluar conexión wiFi (opcional)
189 if (RSSIWiFi < minRSSI) {
190 alertWifi = true;
191 } else {
192 alertWifi = false;
193 }
194 }

290 void updateFromCommand(const String& cmd) {
291 int firstSpace = cmd.indexOf(' ');
292 int secondSpace = cmd.indexOf(' ', firstSpace + 1);
293 if (firstSpace == -1 || secondSpace == -1) return;
294
295 String tipo = cmd.substring(firstSpace + 1, secondSpace);
296 float valor = cmd.substring(secondSpace + 1).toFloat();
297
298 if (tipo == "suelo_min") minSoilMoisture = valor;
299 else if (tipo == "suelo_max") maxSoilMoisture = valor;
300 else if (tipo == "temp_min") minTemperature = valor;
301 else if (tipo == "temp_max") maxTemperature = valor;
302 else if (tipo == "co2_max") maxCO2 = valor;
303 else if (tipo == "luz_min") minLight = (int)valor;
304 else if (tipo == "voltaje_min") minVoltage = valor;
305 else if (tipo == "rssi_min") minRSSI = (int)valor;
306 }
